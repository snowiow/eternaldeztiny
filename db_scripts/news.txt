 CREATE TABLE news (
   id INT NOT NULL auto_increment,
   author VARCHAR(100) NOT NULL,
   title VARCHAR(100) NOT NULL,
   content TEXT NOT NULL,
   date_posted TIMESTAMP NOT NULL,
   PRIMARY KEY (id)
 );
