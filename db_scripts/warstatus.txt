CREATE TABLE warstatus (
    id INT NOT NULL,
    opted_out_date TIMESTAMP NOT NULL,
    opted_in_date TIMESTAMP NOT NULL,
    gemable TINYINT(1) NOT NULL,
    crusade TINYINT(1) NOT NULL,
    reason VARCHAR(255),
    PRIMARY KEY (id),
    FOREIGN KEY (id) REFERENCES account(id) ON DELETE CASCADE
);
